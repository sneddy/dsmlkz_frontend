```python
    # 1. –ø–µ—Ä–≤—ã–µ K-1 –∫–æ–Ω—Ç—Ä–∞—Å—Ç–æ–≤
    cur = sum((a[i] - a[i + 1]) ** 2 for i in range(K - 1))
    best = cur

    # 2. —Å–∫–æ–ª—å–∑—è—â–µ–µ –æ–∫–Ω–æ
    for left in range(1, N - K + 1):
        cur -= (a[left - 1] - a[left]) ** 2           # —É—à—ë–ª —Å–ª–µ–≤–∞
        cur += (a[left + K - 2] - a[left + K - 1]) ** 2  # –¥–æ–±–∞–≤–∏–ª—Å—è —Å–ø—Ä–∞–≤–∞
        if cur > best:
            best = cur
    return best
```

##  –ó–∞–¥–∞—á–∞ B. Binary Classifier

–ê–±–¥–∏–∫–æ–∂–∞ —Å–æ–±—Ä–∞–ª $N$ —Ñ–æ—Ç–æ–≥—Ä–∞—Ñ–∏–π –¥—Ä—É–∑–µ–π, –¥–ª—è –∫–∞–∂–¥–æ–π –∏–∑ –∫–æ—Ç–æ—Ä—ã—Ö –æ–Ω –æ—Ç–º–µ—Ç–∏–ª, —É–ª—ã–±–∞–µ—Ç—Å—è –ª–∏ —á–µ–ª–æ–≤–µ–∫ (–º–µ—Ç–∫–∞ $Y_i = 1$) –∏–ª–∏ –Ω–µ—Ç ($Y_i = 0$). –ó–∞—Ç–µ–º –æ–Ω –∑–∞–ø—É—Å—Ç–∏–ª –±–∏–Ω–∞—Ä–Ω—ã–π –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ç–æ—Ä, –∫–æ—Ç–æ—Ä—ã–π –≤—ã–¥–∞–ª –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å $P_i \in (0.0, 1.0)$ —Ç–æ–≥–æ, —á—Ç–æ —á–µ–ª–æ–≤–µ–∫ –Ω–∞ —Ñ–æ—Ç–æ —É–ª—ã–±–∞–µ—Ç—Å—è.

–ü–æ—Å–ª–µ —ç—Ç–æ–≥–æ –æ–Ω –≤—ã–±—Ä–∞–ª –Ω–µ–∫–æ—Ç–æ—Ä—ã–π **–ø–æ—Ä–æ–≥** $T$ –∏ –æ—Å—Ç–∞–≤–∏–ª —Ç–æ–ª—å–∫–æ —Ç–µ —Ñ–æ—Ç–æ–≥—Ä–∞—Ñ–∏–∏, –¥–ª—è –∫–æ—Ç–æ—Ä—ã—Ö $P_i \ge T$. –û–Ω –∑–∞–º–µ—Ç–∏–ª, —á—Ç–æ –ø—Ä–∏ –∏–∑–º–µ–Ω–µ–Ω–∏–∏ –ø–æ—Ä–æ–≥–∞ $T$ —Ç–æ—á–Ω–æ—Å—Ç—å (precision) –º–æ–∂–µ—Ç **—É–º–µ–Ω—å—à–∞—Ç—å—Å—è** ‚Äî —Ç–æ –µ—Å—Ç—å –º–æ–¥–µ–ª—å —Å—Ç–∞–Ω–æ–≤–∏—Ç—Å—è –º–µ–Ω–µ–µ —É–≤–µ—Ä–µ–Ω–Ω–æ–π. –ê–±–¥–∏–∫–æ–∂–∞ —Ä–µ—à–∏–ª –Ω–∞–π—Ç–∏ –≤—Å–µ —Ç–∞–∫–∏–µ **–∑–Ω–∞—á–µ–Ω–∏—è –ø–æ—Ä–æ–≥–∞ $T$**, –≤ –∫–æ—Ç–æ—Ä—ã—Ö —Ñ—É–Ω–∫—Ü–∏—è $\text{PRECISION}(T)$ —É–±—ã–≤–∞–µ—Ç.

---

#### –û–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è

- **True Positive (TP)** ‚Äî —á–∏—Å–ª–æ –æ—Ç–æ–±—Ä–∞–Ω–Ω—ã—Ö —Ñ–æ—Ç–æ ($P_i \ge T$), –≥–¥–µ $Y_i = 1$
- **False Positive (FP)** ‚Äî —á–∏—Å–ª–æ –æ—Ç–æ–±—Ä–∞–Ω–Ω—ã—Ö —Ñ–æ—Ç–æ ($P_i \ge T$), –≥–¥–µ $Y_i = 0$
- **Precision:**  
  $$
  \text{PRECISION}(T) = \frac{TP}{TP + FP}
  $$

---
## –†–µ—à–µ–Ω–∏–µ

### –ò–¥–µ—è

- –¢–æ—á–Ω–æ—Å—Ç—å –º–µ–Ω—è–µ—Ç—Å—è —Ç–æ–ª—å–∫–æ –≤ —Ç–æ—á–∫–∞—Ö $T = P_i$
- –°–æ—Ä—Ç–∏—Ä—É–µ–º –ø–∞—Ä—ã $(Y_i, P_i)$ –ø–æ —É–±—ã–≤–∞–Ω–∏—é $P_i$
- –ü–æ–¥–¥–µ—Ä–∂–∏–≤–∞–µ–º TP, FP –∏ —Å—á–∏—Ç–∞–µ–º —Ç–æ—á–Ω–æ—Å—Ç—å –¥–æ –∏ –ø–æ—Å–ª–µ –≤–∫–ª—é—á–µ–Ω–∏—è –Ω–æ–≤–æ–π —Ñ–æ—Ç–æ–≥—Ä–∞—Ñ–∏–∏
- –ï—Å–ª–∏ $\text{after} < \text{before}$ ‚Äî —Å–æ—Ö—Ä–∞–Ω—è–µ–º $T = P_i$

---

### 3. –ê–ª–≥–æ—Ä–∏—Ç–º

```text
1. –°—á–∏—Ç–∞—Ç—å –ø–∞—Ä—ã (Y_i, P_i)
2. –û—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞—Ç—å –ø–æ —É–±—ã–≤–∞–Ω–∏—é P_i
3. –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä–æ–≤–∞—Ç—å: tp = 0, fp = 0, prev = -1
4. –î–ª—è –∫–∞–∂–¥–æ–π (Y_i, P_i):
     - precision_before = tp / (tp + fp)
     - –µ—Å–ª–∏ Y_i == 1 ‚Üí tp += 1, –∏–Ω–∞—á–µ ‚Üí fp += 1
     - precision_after = tp / (tp + fp)
     - –µ—Å–ª–∏ after < before ‚Üí –¥–æ–±–∞–≤–∏—Ç—å P_i –≤ –æ—Ç–≤–µ—Ç
     - prev = after
```

## –ó–∞–¥–∞—á–∞ –° - CPFED Net

–í CPFED –∑–∞–¥—É–º–∞–ª–∏—Å—å, –º–æ–∂–Ω–æ –ª–∏ –ø–æ—Å—Ç—Ä–æ–∏—Ç—å –∏—Å–∫—É—Å—Å—Ç–≤–µ–Ω–Ω—ã–π –∏–Ω—Ç–µ–ª–ª–µ–∫—Ç –¥–ª—è –ø–æ–º–æ—â–∏ —Å–∏—Å—Ç–µ–º–µ –ø—Ä–æ–∫—Ç–æ—Ä–∏–Ω–≥–∞. –î–ª—è —ç—Ç–æ–≥–æ —Ä–∞–∑—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç—Å—è –Ω–µ–π—Ä–æ—Å–µ—Ç—å **CPFED Net**, –ø–æ—Å—Ç—Ä–æ–µ–Ω–Ω–∞—è –ø–æ –ø—Ä–∏–Ω—Ü–∏–ø—É –º–Ω–æ–≥–æ—Å–ª–æ–π–Ω–æ–π –ª–∏–Ω–µ–π–Ω–æ–π —Ä–µ–≥—Ä–µ—Å—Å–∏–∏.

–ö–∞–∂–¥—ã–π —Å–ª–æ–π —Å–æ–¥–µ—Ä–∂–∏—Ç $n$ –Ω–µ–π—Ä–æ–Ω–æ–≤, –≤—Å–µ–≥–æ —Å–ª–æ—ë–≤ ‚Äî $k$. –í—Å–µ –Ω–µ–π—Ä–æ–Ω—ã –≤–Ω—É—Ç—Ä–∏ –æ–¥–Ω–æ–≥–æ —Å–ª–æ—è –ø–æ–ª—É—á–∞—é—Ç –æ–¥–∏–Ω –∏ —Ç–æ—Ç –∂–µ –≤—Ö–æ–¥ ‚Äî —Å—É–º–º–∞—Ä–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ —Å –ø—Ä–µ–¥—ã–¥—É—â–µ–≥–æ —Å–ª–æ—è (–Ω–∞ –ø–µ—Ä–≤–æ–º —Å–ª–æ–µ ‚Äî —ç—Ç–æ –ø—Ä–æ—Å—Ç–æ –≤—Ö–æ–¥–Ω–æ–π —Å–∏–≥–Ω–∞–ª $x$).

–ò—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è —Ç–µ—Ö–Ω–∏–∫–∞ **Dropout** ‚Äî —Å–ª—É—á–∞–π–Ω–æ–µ "–æ—Ç–∫–ª—é—á–µ–Ω–∏–µ" –Ω–µ–π—Ä–æ–Ω–æ–≤:
- —Å –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å—é $p$ –Ω–µ–π—Ä–æ–Ω –≤—ã–¥–∞—ë—Ç –∑–Ω–∞—á–µ–Ω–∏–µ $0.0$;
- —Å –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å—é $(1 - p)$ –Ω–µ–π—Ä–æ–Ω –≤—ã–¥–∞—ë—Ç:
  $$
  y_j = a_j \cdot x + b_j
  $$

Dropout –ø—Ä–∏–º–µ–Ω—è–µ—Ç—Å—è **–Ω–µ–∑–∞–≤–∏—Å–∏–º–æ** –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –Ω–µ–π—Ä–æ–Ω–∞.

–§–∏–Ω–∞–ª—å–Ω—ã–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç —Å–µ—Ç–∏ ‚Äî —Å—É–º–º–∞ –≤—ã—Ö–æ–¥–æ–≤ **–≤—Å–µ—Ö –Ω–µ–π—Ä–æ–Ω–æ–≤ –ø–æ—Å–ª–µ–¥–Ω–µ–≥–æ —Å–ª–æ—è**.

---

### –í–∞—à–∞ –∑–∞–¥–∞—á–∞

–í—ã—á–∏—Å–ª–∏—Ç–µ **–º–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –æ–∂–∏–¥–∞–Ω–∏–µ** —Ñ–∏–Ω–∞–ª—å–Ω–æ–≥–æ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞ –Ω–µ–π—Ä–æ—Å–µ—Ç–∏.

**–ß—Ç–æ —Ç–∞–∫–æ–µ –º–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –æ–∂–∏–¥–∞–Ω–∏–µ?**  
–≠—Ç–æ —Å—Ä–µ–¥–Ω–µ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –≤—ã—Ö–æ–¥–Ω–æ–π –≤–µ–ª–∏—á–∏–Ω—ã, –µ—Å–ª–∏ —ç–∫—Å–ø–µ—Ä–∏–º–µ–Ω—Ç (–æ—Ç–∫–ª—é—á–µ–Ω–∏–µ –Ω–µ–π—Ä–æ–Ω–æ–≤) –ø–æ–≤—Ç–æ—Ä—è–µ—Ç—Å—è –±–µ—Å–∫–æ–Ω–µ—á–Ω–æ–µ —á–∏—Å–ª–æ —Ä–∞–∑.

–ï—Å–ª–∏ –Ω–µ–π—Ä–æ–Ω —Ä–∞–±–æ—Ç–∞–µ—Ç —Å –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å—é $(1 - p)$, –µ–≥–æ –æ–∂–∏–¥–∞–µ–º—ã–π –≤–∫–ª–∞–¥:
$$
\mathbb{E}[y_j] = (1 - p)(a_j \cdot x + b_j)
$$

### –§–æ—Ä–º–∞—Ç –≤–≤–æ–¥–∞

–í –ø–µ—Ä–≤–æ–π —Å—Ç—Ä–æ–∫–µ —á–µ—Ç—ã—Ä–µ –∑–Ω–∞—á–µ–Ω–∏—è:
- $x$ ‚Äî –≤—Ö–æ–¥–Ω–æ–π —Å–∏–≥–Ω–∞–ª (–≤–µ—â–µ—Å—Ç–≤–µ–Ω–Ω–æ–µ —á–∏—Å–ª–æ)
- $k$ ‚Äî —á–∏—Å–ª–æ —Å–ª–æ—ë–≤ $(1 \le k \le 4)$
- $n$ ‚Äî —á–∏—Å–ª–æ –Ω–µ–π—Ä–æ–Ω–æ–≤ –≤ –∫–∞–∂–¥–æ–º —Å–ª–æ–µ $(2 \le n \le 10^5)$
- $p$ ‚Äî –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å –æ—Ç–∫–ª—é—á–µ–Ω–∏—è –Ω–µ–π—Ä–æ–Ω–∞ $(0 \le p \le 1)$

–î–∞–ª–µ–µ $k \cdot n$ —Å—Ç—Ä–æ–∫ ‚Äî –ø–æ –æ–¥–Ω–æ–π –Ω–∞ –∫–∞–∂–¥—ã–π –Ω–µ–π—Ä–æ–Ω, —Å–Ω–∞—á–∞–ª–∞ $n$ –Ω–µ–π—Ä–æ–Ω–æ–≤ –ø–µ—Ä–≤–æ–≥–æ —Å–ª–æ—è, –∑–∞—Ç–µ–º –≤—Ç–æ—Ä–æ–≥–æ –∏ —Ç.–¥.  
–ö–∞–∂–¥–∞—è —Å—Ç—Ä–æ–∫–∞ —Å–æ–¥–µ—Ä–∂–∏—Ç –¥–≤–∞ —Ü–µ–ª—ã—Ö —á–∏—Å–ª–∞ $a_i$ –∏ $b_i$ $(‚àí50 \le a_i, b_i \le 50)$ ‚Äî –ø–∞—Ä–∞–º–µ—Ç—Ä—ã –ª–∏–Ω–µ–π–Ω–æ–π —Ä–µ–≥—Ä–µ—Å—Å–∏–∏ –Ω–µ–π—Ä–æ–Ω–∞.

---

### –§–æ—Ä–º–∞—Ç –≤—ã–≤–æ–¥–∞

–û–¥–Ω–æ –≤–µ—â–µ—Å—Ç–≤–µ–Ω–Ω–æ–µ —á–∏—Å–ª–æ ‚Äî –º–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –æ–∂–∏–¥–∞–Ω–∏–µ —Ñ–∏–Ω–∞–ª—å–Ω–æ–≥–æ –≤—ã—Ö–æ–¥–∞ —Å–µ—Ç–∏.  
–û—Ç–≤–µ—Ç —Å—á–∏—Ç–∞–µ—Ç—Å—è –∫–æ—Ä—Ä–µ–∫—Ç–Ω—ã–º, –µ—Å–ª–∏ –∞–±—Å–æ–ª—é—Ç–Ω–∞—è –∏–ª–∏ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–∞—è –ø–æ–≥—Ä–µ—à–Ω–æ—Å—Ç—å –Ω–µ –ø—Ä–µ–≤—ã—à–∞–µ—Ç $10^{-6}$.

## –†–µ—à–µ–Ω–∏–µ

–î–ª—è –∫–∞–∂–¥–æ–≥–æ —Å–ª–æ—è $L$:

$$
y_{l,j} =
\begin{cases}
0, & \text{—Å –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å—é } p \\[4pt]
a_{l,j} \cdot z_{l-1} + b_{l,j}, & \text{—Å –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å—é } 1 - p
\end{cases}
$$

$$
z_{l-1} = \sum_{j=1}^{n} y_{l-1,j}
\quad\Rightarrow\quad
\mathbb{E}[z_l] = (1 - p) \left( \left( \sum_{j=1}^{n} a_{l,j} \right) \cdot \mathbb{E}[z_{l-1}] + \sum_{j=1}^{n} b_{l,j} \right)
$$

–ë–∞–∑–∞: $z_0 = x$  
–ò—Å–∫–æ–º–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ: $\mathbb{E}[z_k]$

---

### 2. –ü—Å–µ–≤–¥–æ–∫–æ–¥

1. –°—á–∏—Ç–∞—Ç—å $x,\;k,\;n,\;p$  
2. $S \leftarrow x$  
3. –î–ª—è –∫–∞–∂–¥–æ–≥–æ —Å–ª–æ—è:
   - $sumA \leftarrow \sum a_{l,j}$  
   - $sumB \leftarrow \sum b_{l,j}$  
   - $S \leftarrow (1 - p)(sumA \cdot S + sumB)$  
4. –í—ã–≤–µ—Å—Ç–∏ $S$

---

### 3. –°–ª–æ–∂–Ω–æ—Å—Ç—å

- **–í—Ä–µ–º—è:** $O(k \cdot n)$  
- **–ü–∞–º—è—Ç—å:** $O(1)$ (–º–æ–∂–Ω–æ —Å—á–∏—Ç–∞—Ç—å –Ω–∞ –ª–µ—Ç—É)

---

### 4. –ö–æ–¥ (Python)

```python
def main():
    # Read input
    parts = input().split()
    x = float(parts[0])
    k = int(parts[1])
    n = int(parts[2])
    p = float(parts[3])
    
    a = []
    b = []
    for _ in range(k):
        layer_a = []
        layer_b = []
        for _ in range(n):
            coef_a, coef_b = map(int, input().split())
            layer_a.append(coef_a)
            layer_b.append(coef_b)
        a.append(layer_a)
        b.append(layer_b)
    
    prev_layer = x
    for layer in range(k):
        sum_val = 0
        for i in range(n):
            sum_val += a[layer][i] * prev_layer + b[layer][i]
        prev_layer = sum_val * (1 - p)
    
    print(f"{prev_layer:.6f}")
```

### ‚úÖ 1. **MEDIAN PCA** ‚Äî –•–æ—Ä–æ—à–µ–µ —Ä–µ—à–µ–Ω–∏–µ

- –ª–æ–≥–∏—Å—Ç–∏—á–µ—Å–∫–∞—è —Ä–µ–≥—Ä–µ—Å—Å–∏—è –ø–æ –∏–∑–≤–ª–µ—á—ë–Ω–Ω—ã–º –ø—Ä–∏–∑–Ω–∞–∫–∞–º

```python
import numpy as np
from scipy.ndimage import median_filter
from sklearn.decomposition import PCA
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import accuracy_score
from sklearn.model_selection import train_test_split

# --- –ú–µ–¥–∏–∞–Ω–Ω–∞—è —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏—è –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π ---
def denoise_with_median(X, size=2):
    X_denoised = np.zeros_like(X)
    for i in range(X.shape[0]):
        img = X[i].reshape(28, 28)
        filtered = median_filter(img, size=size)
        X_denoised[i] = filtered.flatten()
    return X_denoised

# --- –≠–∫—Å—Ç—Ä–∞–∫—Ç–æ—Ä –ø—Ä–∏–∑–Ω–∞–∫–æ–≤: –º–µ–¥–∏–∞–Ω–Ω–∞—è —Ñ–∏–ª—å—Ç—Ä–∞—Ü–∏—è + PCA ---
class PCADenoizedFeatureExtractor:
    def __init__(self, pca_components: int):
        self.pca_components = pca_components
        self.pca = PCA(n_components=self.pca_components)

    def fit(self, X):
        self.pca.fit(X / 255.0)

    def transform(self, X):
        denoized = denoise_with_median(X, size=2)
        return self.pca.transform(denoized / 255.0)

# --- –û—Å–Ω–æ–≤–Ω–æ–π –ø–∞–π–ø–ª–∞–π–Ω —Ä–µ—à–µ–Ω–∏—è ---
def denoize_pca_solution(X_train, X_test, y_train):
    feature_extractor = PCADenoizedFeatureExtractor(pca_components=20)
    feature_extractor.fit(X_train)
    train_features = feature_extractor.transform(X_train)
    test_features = feature_extractor.transform(X_test)

    model = LogisticRegression(multi_class="multinomial", solver="lbfgs", max_iter=5000)
    model.fit(train_features, y_train)
    y_pred = model.predict(test_features)
    return y_pred

def main(X, y):
    """–î–ª—è –∫—Ä–∞—Ç–∫–æ—Å—Ç–∏ –ø—Ä–æ–ø—É—Å—Ç–∏–º —á–∞—Å—Ç—å —Å –∑–∞–≥—Ä—É–∑–∫–æ–π –¥–∞–Ω–Ω—ã—Ö"""

    # –†–∞–∑–¥–µ–ª–µ–Ω–∏–µ –Ω–∞ train/test
    X_train, X_test, y_train, y_test = train_test_split(
        X, y, test_size=0.5, random_state=42
    )

    # –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–µ –∏ –º–µ—Ç—Ä–∏–∫–∞
    y_pred = denoize_pca_solution(X_train, X_test, y_train)
    acc = accuracy_score(y_test, y_pred)
    print(f"Denoize PCA solution accuracy: {acc:.4f}")
```

```python
class ResizeFeatureExtractor:
    def __init__(self, n_rows: int, n_cols: int):
        self.n_rows = n_rows
        self.n_cols = n_cols
        self.original_size = (28, 28)

    def fit(self, X, y=None):
        return self

    def transform(self, X):
        n_samples = X.shape[0]
        resized = np.zeros((n_samples, self.n_rows * self.n_cols), dtype=np.float32)

        zoom_factors = (
            self.n_rows / self.original_size[0],
            self.n_cols / self.original_size[1],
        )

        for i in range(n_samples):
            img = X[i].reshape(self.original_size)
            img_resized = zoom(img, zoom_factors, order=3)  # bi-cubic interpolation
            resized[i] = img_resized.flatten()

        return resized

def resize_solution(X_train, X_test, y_train):
    # –û–±—É—á–∞–µ–º
    resize_feature_extractor = ResizeFeatureExtractor(5, 4)
    resize_feature_extractor.fit(X_train)
    train_features = resize_feature_extractor.transform(denoise_with_median(X_train))
    test_features = resize_feature_extractor.transform(denoise_with_median(X_test))

    model = LogisticRegression(multi_class="multinomial", solver="lbfgs", max_iter=5000)
    model.fit(train_features, y_train)
    y_pred = model.predict(test_features)
    return y_pred

# --- Main function for this solution ---
def main(X, y):
    """–î–ª—è –∫—Ä–∞—Ç–∫–æ—Å—Ç–∏ –ø—Ä–æ–ø—É—Å—Ç–∏–º —á–∞—Å—Ç—å —Å –∑–∞–≥—Ä—É–∑–∫–æ–π –¥–∞–Ω–Ω—ã—Ö"""
    X_train, X_test, y_train, y_test = train_test_split(
        X, y, test_size=0.5, random_state=42
    )
    y_pred_resize = resize_solution(X_train, X_test, y_train)

    acc_resize = accuracy_score(y_test, y_pred_resize)
    print(f"Resize solution: {acc_resize:.4f}")
```

```python
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import accuracy_score, classification_report

from nltk.stem.snowball import SnowballStemmer
import re

data = pd.read_csv("arxiv_data/arxiv_dataset.csv", index_col=0)

X_train = data.loc[:2999, 'sampled_sentence']
X_test = data.loc[3000:, 'sampled_sentence']
y_train = data.loc[:2999, 'paper_section']

# –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º —Å—Ç–µ–º–º–µ—Ä
stemmer = SnowballStemmer("english")

# –ö–∞—Å—Ç–æ–º–Ω–∞—è —Ç–æ–∫–µ–Ω–∏–∑–∞—Ü–∏—è + —Å—Ç–µ–º–º–∏–Ω–≥
def tokenize_and_stem(text):
    # –ë–∞–∑–æ–≤–∞—è –Ω–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è —Ç–µ–∫—Å—Ç–∞

    text = re.sub(r'\d+', '', text)
    text = re.sub(r'[^\w\s]', '', text)
    tokens = text.strip().split()
    return [stemmer.stem(token) for token in tokens if len(token) > 2]

    tokenizer=tokenize_and_stem,
    ngram_range=(1, 2),          # unigrams + bigrams
    stop_words='english',        # —É–±–∏—Ä–∞–µ–º —á–∞—Å—Ç—ã–µ —Å–ª–æ–≤–∞
    max_df=0.06,                 # —É–±–∏—Ä–∞–µ–º —Å—É–ø–µ—Ä–≤—Å—Ç—Ä–µ—á–∞—é—â–∏–µ—Å—è —Å–ª–æ–≤–∞
)

X_train_vec = vectorizer.fit_transform(X_train)  # sparse matrix
X_test_vec = vectorizer.transform(X_test)

model = LogisticRegression(max_iter=1000)
model.fit(X_train_vec, y_train)

y_pred = model.predict(X_test_vec)
```
